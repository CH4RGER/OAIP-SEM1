bool is_num(string a) { 
bool res = false; 
if (a[0] >= '0' && a[0] <= '9') 
res = true; 
return res; 
} 

bool is_sign(string a) { 
bool res = false; 
if (a[0] == '+' || a[0] == '-') 
res = true; 
return res; 
}
