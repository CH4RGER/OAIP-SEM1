#include <iostream>
#include <string>
#include <clocale>

using namespace std;

void main()
{
	setlocale(LC_ALL, "Russian");

	string str;
	char comma = ',';
	int qty = 0;

	getline(cin, str);

	for (int i = 0; i < str.length(); i++)
		if (str[i] == comma)
			qty++;

	if (qty > 3) {
		cout << "Какую запятую по счету менят: ";
		cin >> qty;
		for (int i = 0; i < str.length(); i++) {
			if (str[i] == comma) qty--;
			if (!qty) {
				str.replace(i, 1 , ".");
				break;
			}
		}

	}
	else
		for (int i = 0; i < str.length(); i++)
			if (str[i] == comma)
				str.replace(i, 1, " ") ;

	cout << str << endl;
	system("pause");
}
